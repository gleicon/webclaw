Replace the stub in static/local-llm-worker.js with the full implementation. The worker auto-selects WebLLM (WebGPU) or wllama (CPU) based on the `backend` field in the LOAD_MODEL payload.

Key implementation points:
- Import WebLLM via the npm ESM bundle. Because this worker runs in a Web Worker context, use the bundle path: the Vite build will resolve `@mlc-ai/web-llm`. In dev mode with Make serve (Go devserver), the import must work via a `/vendor/` path — add a note but keep the import as-is since Vite handles bundling.
- Import wllama via `/vendor/wllama.js` path (to be set up in plan 10-03's vite.config.js update).
- For CHAT_STREAM with WebLLM: use `engine.chat.completions.create({ stream: true })` OpenAI-compatible API, forwarding each delta as TOKEN message.
- For CHAT_STREAM with wllama: use `wllama.createCompletion(prompt, { onNewToken })` pattern.
- Call `navigator.storage.persist()` after successful model load to request persistent Cache API storage.
- Catch all errors and send ERROR message rather than throwing, so the host can display them.

```javascript
// static/local-llm-worker.js
// Full implementation — WebLLM (WebGPU) + wllama (CPU fallback)
// See .planning/phases/10-browser-local-model/10-RESEARCH.md for API references

// Dynamic import allows the worker to work before Vite bundling
// The actual path depends on build mode; plan 10-03 adds vite.config.js vendor copying.
let MLCEngine = null;
let Wllama = null;

async function loadInferenceLib(backend) {
    if (backend === 'webgpu') {
        const mod = await import('@mlc-ai/web-llm');
        MLCEngine = mod.MLCEngine;
    } else {
        const mod = await import('/vendor/wllama.js');
        Wllama = mod.Wllama;
    }
}

let engine = null;
let wllamaInstance = null;
let currentBackend = null;
let cancelled = false;

self.onmessage = async function(event) {
    const { type, payload } = event.data;

    if (type === 'CANCEL') {
        cancelled = true;
        engine = null;
        wllamaInstance = null;
        return;
    }

    if (type === 'LOAD_MODEL') {
        cancelled = false;
        const { modelId, backend } = payload;
        currentBackend = backend;

        try {
            await loadInferenceLib(backend);

            if (backend === 'webgpu') {
                engine = new MLCEngine();
                await engine.reload(modelId, {
                    initProgressCallback: (report) => {
                        self.postMessage({
                            type: 'LOAD_PROGRESS',
                            payload: { progress: report.progress, text: report.text }
                        });
                    }
                });
            } else {
                // CPU path via wllama
                wllamaInstance = new Wllama({
                    'single-thread/wllama.wasm': '/vendor/wllama-single.wasm',
                    'multi-thread/wllama.wasm': '/vendor/wllama-multi.wasm',
                });
                await wllamaInstance.loadModelFromUrl(modelId, {
                    progressCallback: ({ loaded, total }) => {
                        const pct = total > 0 ? loaded / total : 0;
                        self.postMessage({
                            type: 'LOAD_PROGRESS',
                            payload: { progress: pct, text: `Downloading model... ${Math.round(pct * 100)}%` }
                        });
                    }
                });
            }

            // Request persistent storage to prevent model eviction
            if (navigator.storage && navigator.storage.persist) {
                navigator.storage.persist().then(granted => {
                    if (!granted) {
                        self.postMessage({
                            type: 'LOAD_PROGRESS',
                            payload: { progress: 1.0, text: 'Warning: storage persistence denied — model may be re-downloaded on next visit' }
                        });
                    }
                });
            }

            self.postMessage({
                type: 'MODEL_READY',
                payload: { modelId, backend }
            });

        } catch (err) {
            self.postMessage({
                type: 'ERROR',
                payload: { message: `Failed to load model: ${err.message}` }
            });
        }
        return;
    }

    if (type === 'CHAT_STREAM') {
        if (!engine && !wllamaInstance) {
            self.postMessage({ type: 'ERROR', payload: { message: 'No model loaded. Send LOAD_MODEL first.' } });
            return;
        }

        const { messages, temperature = 0.7, maxTokens = 512 } = payload;
        cancelled = false;
        let totalTokens = 0;

        try {
            if (currentBackend === 'webgpu' && engine) {
                const stream = await engine.chat.completions.create({
                    messages,
                    stream: true,
                    temperature,
                    max_tokens: maxTokens,
                });
                for await (const chunk of stream) {
                    if (cancelled) break;
                    const delta = chunk.choices[0]?.delta?.content ?? '';
                    if (delta) {
                        totalTokens++;
                        self.postMessage({ type: 'TOKEN', payload: { text: delta } });
                    }
                }
            } else if (wllamaInstance) {
                // wllama uses a prompt string; build from messages
                const prompt = messages.map(m => `${m.role}: ${m.content}`).join('\n') + '\nassistant:';
                await wllamaInstance.createCompletion(prompt, {
                    nPredict: maxTokens,
                    temperature,
                    onNewToken: (token, piece) => {
                        if (cancelled) return;
                        totalTokens++;
                        self.postMessage({ type: 'TOKEN', payload: { text: piece } });
                    }
                });
            }

            if (!cancelled) {
                self.postMessage({ type: 'COMPLETE', payload: { totalTokens } });
            }
        } catch (err) {
            self.postMessage({ type: 'ERROR', payload: { message: `Inference error: ${err.message}` } });
        }
        return;
    }
};
```

```javascript
// WebGPU detection with graceful fallback
// Source: MDN WebGPU API + webgpu-webllm-app pattern (see RESEARCH.md)
async function detectLocalMLBackend() {
    if (!('gpu' in navigator)) return 'cpu';
    try {
        const adapter = await navigator.gpu.requestAdapter();
        if (!adapter) return 'cpu';
        return 'webgpu';
    } catch (e) {
        return 'cpu';
    }
}
```

```javascript
async function initLocalModel(modelId) {
    if (!modelId) return;

    const backend = window.webclaw.localML.backend || await detectLocalMLBackend();
    window.webclaw.localML.backend = backend;
    window.webclaw.localML.loading = true;

    // Spawn dedicated worker (separate from worker.js which runs WASM agent loop)
    const localWorker = new Worker('/static/local-llm-worker.js', { type: 'module' });
    window.webclaw.localML.worker = localWorker;

    localWorker.onmessage = function(e) {
        const { type, payload } = e.data;

        if (type === 'LOAD_PROGRESS') {
            // Forward progress to Go and to any UI listeners
            window.dispatchEvent(new CustomEvent('webclaw:local-model-progress', {
                detail: { progress: payload.progress, text: payload.text }
            }));
            // Also call Go bridge callback if registered
            if (window.webclaw.local && window.webclaw.local.onProgress) {
                window.webclaw.local.onProgress(payload.progress, payload.text || '');
            }
        }

        if (type === 'MODEL_READY') {
            window.webclaw.localML.ready = true;
            window.webclaw.localML.loading = false;
            window.webclaw.localML.modelId = payload.modelId;
            window.dispatchEvent(new CustomEvent('webclaw:local-model-ready', {
                detail: { modelId: payload.modelId, backend: payload.backend }
            }));
            // Signal Go that model is ready
            if (window.webclaw.local && window.webclaw.local.onComplete) {
                // onComplete signals readiness in the bridge
            }
        }

        if (type === 'TOKEN') {
            if (window.webclaw.local && window.webclaw.local.onToken) {
                window.webclaw.local.onToken(payload.text);
            }
        }

        if (type === 'COMPLETE') {
            if (window.webclaw.local && window.webclaw.local.onComplete) {
                window.webclaw.local.onComplete();
            }
        }

        if (type === 'ERROR') {
            window.webclaw.localML.loading = false;
            window.dispatchEvent(new CustomEvent('webclaw:local-model-error', {
                detail: { message: payload.message }
            }));
            if (window.webclaw.local && window.webclaw.local.onError) {
                window.webclaw.local.onError(payload.message);
            }
        }
    };

    localWorker.postMessage({ type: 'LOAD_MODEL', payload: { modelId, backend } });
}

// streamChat — called by Go LocalProvider via webclaw.local.streamChat
// Forwards inference request to local-llm-worker.js
function streamLocalChat(messagesJSON) {
    const worker = window.webclaw.localML.worker;
    if (!worker || !window.webclaw.localML.ready) {
        if (window.webclaw.local && window.webclaw.local.onError) {
            window.webclaw.local.onError('Local model not ready');
        }
        return;
    }
    let messages;
    try {
        messages = JSON.parse(messagesJSON);
    } catch (e) {
        window.webclaw.local.onError('Invalid messages JSON: ' + e.message);
        return;
    }
    worker.postMessage({
        type: 'CHAT_STREAM',
        payload: { messages, temperature: 0.7, maxTokens: 512 }
    });
}

// Expose initLocalModel and streamLocalChat on window.webclaw for Go bridge and settings UI
window.webclaw.localML.init = initLocalModel;
window.webclaw.localML.stream = streamLocalChat;
```
