```javascript
// test/unit/local-tool-json.test.js
// Node unit test — run with: node test/unit/local-tool-json.test.js
const assert = require('assert');

// Simulate what the local model returns in JSON mode for a memory_search tool call
const VALID_TOOL_OUTPUT = JSON.stringify({
  tool: "memory_search",
  args: { query: "test query" }
});

const INVALID_TOOL_OUTPUT = "I will search your memory for that.";

function parseLocalToolOutput(jsonStr) {
  try {
    const parsed = JSON.parse(jsonStr);
    if (!parsed.tool || typeof parsed.tool !== 'string') return null;
    if (!parsed.args || typeof parsed.args !== 'object') return null;
    return { toolName: parsed.tool, args: parsed.args };
  } catch {
    return null;
  }
}

// Test 1: valid JSON tool output parses correctly
const result = parseLocalToolOutput(VALID_TOOL_OUTPUT);
assert.strictEqual(result.toolName, 'memory_search');
assert.deepStrictEqual(result.args, { query: 'test query' });

// Test 2: plain text returns null (model didn't use JSON mode correctly)
assert.strictEqual(parseLocalToolOutput(INVALID_TOOL_OUTPUT), null);

// Test 3: missing .tool field returns null
assert.strictEqual(parseLocalToolOutput(JSON.stringify({ args: {} })), null);

// Test 4: missing .args field returns null
assert.strictEqual(parseLocalToolOutput(JSON.stringify({ tool: 'memory_search' })), null);

console.log('All local-tool-json tests passed.');
```

```go
package provider

import (
	"context"
	"fmt"

// LocalProvider delegates LLM inference to a JS-side WebLLM/wllama engine.
// The actual inference runs in static/local-llm-worker.js via postMessage.
// Tokens are forwarded back to Go via callbacks registered in jsbridge/local_bridge.go.
//
// Integration: Register as "local" vendor in Router from cmd/webclaw/main.go.
// Model IDs use the full WebLLM format: "local/Llama-3.2-3B-Instruct-q4f16_1-MLC"
type LocalProvider struct {
	modelID string
	ready   bool
}

// NewLocalProvider creates a new local provider stub.
// The provider is not ready until the JS worker signals MODEL_READY.
func NewLocalProvider(modelID string) *LocalProvider {
	return &LocalProvider{
		modelID: modelID,
		ready:   false,
	}
}

// Name returns the provider identifier used in routing (vendor prefix).
func (p *LocalProvider) Name() string { return "local" }

// MaxContextWindow returns the context window for local models.

func (p *LocalProvider) MaxContextWindow(model string) int { return 4096 }

// GetModel returns the configured model ID for ProviderChain compatibility.
func (p *LocalProvider) GetModel() string { return p.modelID }

// SetReady marks the provider as ready after MODEL_READY signal from worker.
```

Create `internal/jsbridge/local_bridge.go` — a stub that exposes `webclaw.local` on the JS global. Pattern follows `RegisterOAuthBridge` from oauth_bridge.go exactly:

```go
//go:build js && wasm

package jsbridge

import "syscall/js"

// RegisterLocalModelBridge exposes Go callbacks for the local-llm-worker.js.
// Pattern mirrors RegisterOAuthBridge from oauth_bridge.go.
//
// Exports to window.webclaw.local:
//   - onToken(text string)    — called by JS for each streamed token
//   - onComplete()            — called by JS when generation finishes
//   - onError(msg string)     — called by JS on inference error
//   - onProgress(pct float64, text string) — called during model load
//   - streamChat(messagesJSON string) — called by Go to start inference
func RegisterLocalModelBridge(
	onToken func(text string),
	onComplete func(),
	onError func(msg string),
	onProgress func(progress float64, text string),
) {
	webclaw := js.Global().Get("webclaw")
	if webclaw.IsUndefined() || webclaw.IsNull() {
		return
	}

	obj := js.Global().Get("Object").New()

	obj.Set("onToken", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
		if len(args) > 0 {
			onToken(args[0].String())
		}
		return nil
	}))

	obj.Set("onComplete", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
		onComplete()
		return nil
	}))

	obj.Set("onError", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
		if len(args) > 0 {
			onError(args[0].String())
		}
		return nil
	}))

	obj.Set("onProgress", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
		progress := 0.0
		text := ""
		if len(args) > 0 {
			progress = args[0].Float()
		}
		if len(args) > 1 {
			text = args[1].String()
		}
		onProgress(progress, text)
		return nil
	}))

	// streamChat is called by Go to initiate inference; JS forwards to worker
	// TODO (plan 10-02): webclaw-host.js must implement streamChat forwarding
	obj.Set("streamChat", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
		// Placeholder — webclaw-host.js will implement the actual forwarding
		return nil
	}))

	webclaw.Set("local", obj)
}

// InitLocalModelBridge initializes the local model JS bridge with no-op callbacks.
// Called from main.go to ensure webclaw.local object exists even before model loads.
func InitLocalModelBridge() {
	RegisterLocalModelBridge(
		func(text string) {},
		func() {},
		func(msg string) {},
		func(progress float64, text string) {},
	)
}
```
