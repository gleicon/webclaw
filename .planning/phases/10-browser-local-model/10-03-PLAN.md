**Update internal/jsbridge/local_bridge.go** to add channel management alongside the existing callback registration. The key pattern: Go creates channels, passes them to the bridge, JS callbacks write to the channels, Go reads from channels:

Add to jsbridge/local_bridge.go:
```go
// Global channels for current inference request.
// Only one inference request can be active at a time (local models are single-threaded).
var (
    globalTokenCh  chan string
    globalDoneCh   chan struct{}
    globalErrCh    chan string
)

// SetLocalModelChannels sets the channels for the current inference request.
// Called by LocalProvider.Stream() before calling webclaw.local.streamChat.
func SetLocalModelChannels(tokenCh chan string, doneCh chan struct{}, errCh chan string) {
    globalTokenCh = tokenCh
    globalDoneCh = doneCh
    globalErrCh = errCh
}

// ClearLocalModelChannels clears the channels after inference completes.
func ClearLocalModelChannels() {
    globalTokenCh = nil
    globalDoneCh = nil
    globalErrCh = nil
}
```

Update RegisterLocalModelBridge callbacks to write to these channels:
```go
obj.Set("onToken", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
    if len(args) > 0 && globalTokenCh != nil {
        select {
        case globalTokenCh <- args[0].String():
        default: // non-blocking: skip if channel full
        }
    }
    return nil
}))

obj.Set("onComplete", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
    if globalDoneCh != nil {
        select {
        case globalDoneCh <- struct{}{}:
        default:
        }
    }
    return nil
}))

obj.Set("onError", js.FuncOf(func(this js.Value, args []js.Value) interface{} {
    if len(args) > 0 && globalErrCh != nil {
        select {
        case globalErrCh <- args[0].String():
        default:
        }
    }
    return nil
}))
```

Add CallStreamChat function to jsbridge (called by LocalProvider):
```go
// CallStreamChat invokes webclaw.local.streamChat with messages JSON.
// Must be called after SetLocalModelChannels.
func CallStreamChat(messagesJSON string) error {
    webclaw := js.Global().Get("webclaw")
    if webclaw.IsUndefined() || webclaw.IsNull() {
        return fmt.Errorf("webclaw not initialized")
    }
    local := webclaw.Get("local")
    if local.IsUndefined() || local.IsNull() {
        return fmt.Errorf("webclaw.local not registered")
    }
    streamChat := local.Get("streamChat")
    if streamChat.IsUndefined() || streamChat.IsNull() {
        return fmt.Errorf("webclaw.local.streamChat not available")
    }
    streamChat.Invoke(messagesJSON)
    return nil
}

// IsLocalModelReady checks window.webclaw.localML.ready.
func IsLocalModelReady() bool {
    webclaw := js.Global().Get("webclaw")
    if webclaw.IsUndefined() || webclaw.IsNull() {
        return false
    }
    localML := webclaw.Get("localML")
    if localML.IsUndefined() || localML.IsNull() {
        return false
    }
    return localML.Get("ready").Bool()
}
```

**Update internal/provider/local.go** Stream() to use the channel bridge:

```go
func (p *LocalProvider) Stream(ctx context.Context, req CompletionRequest) <-chan Token {
    outCh := make(chan Token, 32)

    go func() {
        defer close(outCh)

        // Check JS-side readiness
        if !jsbridge.IsLocalModelReady() {
            outCh <- Token{
                FinishReason: "error",
                Text:         "local model not ready — load a model in Settings -> Local Model",
            }
            return
        }

        // Serialize messages to JSON for JS bridge
        msgs := make([]map[string]string, len(req.Messages))
        for i, m := range req.Messages {
            msgs[i] = map[string]string{"role": m.Role, "content": m.Content}
        }
        messagesJSON, err := json.Marshal(msgs)
        if err != nil {
            outCh <- Token{FinishReason: "error", Text: "failed to serialize messages: " + err.Error()}
            return
        }

        // Set up channels for this inference request
        tokenCh := make(chan string, 64)
        doneCh := make(chan struct{}, 1)
        errCh := make(chan string, 1)
        jsbridge.SetLocalModelChannels(tokenCh, doneCh, errCh)
        defer jsbridge.ClearLocalModelChannels()

        // Start inference
        if err := jsbridge.CallStreamChat(string(messagesJSON)); err != nil {
            outCh <- Token{FinishReason: "error", Text: err.Error()}
            return
        }

        // Forward tokens to output channel until done or cancelled
        for {
            select {
            case <-ctx.Done():
                return
            case tok := <-tokenCh:
                outCh <- Token{Text: tok}
            case errMsg := <-errCh:
                outCh <- Token{FinishReason: "error", Text: errMsg}
                return
            case <-doneCh:
                outCh <- Token{FinishReason: "stop"}
                return
            }
        }
    }()

    return outCh
}
```

```go
// Phase 10: Initialize local model JS bridge
// This registers webclaw.local callbacks before WASM signals ready
jsbridge.InitLocalModelBridge()

// Phase 10: Register local provider if configured in localStorage
// Config stored at 'webclaw_local_model_config' (not IndexedDB — avoids Pitfall 3)
go func() {
    // Wait for WASM to be fully initialized before reading localStorage
    time.Sleep(1 * time.Second)

    localConfig := js.Global().Get("localStorage").Call("getItem", "webclaw_local_model_config")
    if !localConfig.IsNull() && !localConfig.IsUndefined() {
        configStr := localConfig.String()
        var config map[string]interface{}
        if err := json.Unmarshal([]byte(configStr), &config); err == nil {
            enabled, _ := config["enabled"].(bool)
            modelID, _ := config["modelId"].(string)
            if enabled && modelID != "" {
                localProv := provider.NewLocalProvider(modelID)
                if globalRouter != nil {
                    globalRouter.RegisterProvider("local", localProv)
                    js.Global().Get("console").Call("log", "[webclaw] local provider registered:", modelID)
                }
                // Signal JS to start loading the model
                webclaw := js.Global().Get("webclaw")
                if !webclaw.IsUndefined() {
                    localML := webclaw.Get("localML")
                    if !localML.IsUndefined() {
                        initFn := localML.Get("init")
                        if !initFn.IsUndefined() {
                            initFn.Invoke(modelID)
                        }
                    }
                }
            }
        }
    }
}()
```
