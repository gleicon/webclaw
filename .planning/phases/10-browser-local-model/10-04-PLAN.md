**Add initialization script at the bottom of the settings tab section** (or in a `<script>` block before `</body>`). This script wires the UI to webclaw.localML events using safe DOM methods:

```javascript
(function initLocalModelUI() {
  // Populate model dropdown using createElement (safe for variable content)
  const models = [
    { id: 'Llama-3.2-3B-Instruct-q4f16_1-MLC', label: 'Llama 3.2 3B (2.2GB, best quality)' },
    { id: 'Gemma-2-2B-it-q4f16_1-MLC', label: 'Gemma 2 2B (1.5GB, smaller download)' },
    { id: 'Qwen2.5-1.5B-Instruct-q4f16_1-MLC', label: 'Qwen 2.5 1.5B (1.0GB, fastest)' },
    { id: 'https://huggingface.co/Qwen/Qwen2.5-1.5B-Instruct-GGUF/resolve/main/qwen2.5-1.5b-instruct-q4_k_m.gguf',
      label: 'Qwen 1.5B GGUF (CPU only, 1.0GB)' },
  ];

  const select = document.getElementById('local-model-select');
  if (select) {
    models.forEach(m => {
      const opt = document.createElement('option');
      opt.value = m.id;
      opt.textContent = m.label;  // textContent — safe, not innerHTML
      select.appendChild(opt);
    });
  }

  // Show backend after detection
  function updateBackendDisplay() {
    const backendEl = document.getElementById('local-ml-backend');
    if (!backendEl) return;
    const backend = window.webclaw?.localML?.backend;
    if (backend === 'webgpu') {
      backendEl.textContent = 'WebGPU (GPU accelerated)';
      backendEl.className = 'text-xs text-green-400';
    } else if (backend === 'cpu') {
      backendEl.textContent = 'CPU (WASM — slower)';
      backendEl.className = 'text-xs text-yellow-400';
    } else {
      backendEl.textContent = 'Detecting...';
    }
  }

  // Storage estimate
  if (navigator.storage && navigator.storage.estimate) {
    navigator.storage.estimate().then(({ quota, usage }) => {
      const available = quota - usage;
      const availGB = (available / 1e9).toFixed(1);
      const el = document.getElementById('local-storage-estimate');
      if (el) {
        el.textContent = `Available storage: ${availGB}GB`;
        if (available < 1.5e9) {
          el.textContent += ' (warning: may not be enough for model)';
          el.className = 'mb-3 text-xs text-yellow-500';
        }
      }
    });
  }

  // Status update helpers
  function setStatus(color, text) {
    const dot = document.getElementById('local-model-status-dot');
    const statusText = document.getElementById('local-model-status-text');
    if (dot) dot.className = `w-2 h-2 rounded-full bg-${color}-500`;
    if (statusText) statusText.textContent = text;
  }

  // Load existing config from localStorage
  function loadConfig() {
    try {
      const raw = localStorage.getItem('webclaw_local_model_config');
      return raw ? JSON.parse(raw) : null;
    } catch { return null; }
  }

  function saveConfig(config) {
    localStorage.setItem('webclaw_local_model_config', JSON.stringify(config));
  }

  // Initialize UI state from saved config
  const savedConfig = loadConfig();
  if (savedConfig && savedConfig.enabled) {
    const select = document.getElementById('local-model-select');
    if (select && savedConfig.modelId) select.value = savedConfig.modelId;
    setStatus('yellow', 'Loading...');
    const enableBtn = document.getElementById('local-model-enable-btn');
    const disableBtn = document.getElementById('local-model-disable-btn');
    if (enableBtn) enableBtn.classList.add('hidden');
    if (disableBtn) disableBtn.classList.remove('hidden');
  }

  // Progress events
  window.addEventListener('webclaw:local-model-progress', function(e) {
    const wrap = document.getElementById('local-model-progress-wrap');
    const bar = document.getElementById('local-model-progress-bar');
    const text = document.getElementById('local-model-progress-text');
    if (wrap) wrap.classList.remove('hidden');
    if (bar) bar.style.width = `${Math.round((e.detail.progress || 0) * 100)}%`;
    if (text) text.textContent = e.detail.text || '';
    setStatus('blue', 'Downloading...');
  });

  window.addEventListener('webclaw:local-model-ready', function(e) {
    const wrap = document.getElementById('local-model-progress-wrap');
    if (wrap) wrap.classList.add('hidden');
    setStatus('green', 'Ready — ' + (e.detail.modelId || '').split('/').pop().slice(0, 30));
  });

  window.addEventListener('webclaw:local-model-error', function(e) {
    const wrap = document.getElementById('local-model-progress-wrap');
    if (wrap) wrap.classList.add('hidden');
    setStatus('red', 'Error: ' + (e.detail.message || 'unknown'));
  });

  // Enable button
  const enableBtn = document.getElementById('local-model-enable-btn');
  if (enableBtn) {
    enableBtn.addEventListener('click', function() {
      const select = document.getElementById('local-model-select');
      const modelId = select ? select.value : '';
      if (!modelId) return;

      const config = {
        enabled: true,
        modelId,
        backend: window.webclaw?.localML?.backend || 'webgpu'
      };
      saveConfig(config);

      enableBtn.classList.add('hidden');
      const disableBtn = document.getElementById('local-model-disable-btn');
      if (disableBtn) disableBtn.classList.remove('hidden');
      setStatus('yellow', 'Loading model...');

      // Trigger model load via webclaw-host.js
      if (window.webclaw?.localML?.init) {
        window.webclaw.localML.init(modelId);
      }
    });
  }

  // Disable button
  const disableBtn = document.getElementById('local-model-disable-btn');
  if (disableBtn) {
    disableBtn.addEventListener('click', function() {
      saveConfig({ enabled: false, modelId: '', backend: '' });
      setStatus('gray', 'Disabled');
      enableBtn.classList.remove('hidden');
      disableBtn.classList.add('hidden');
      const wrap = document.getElementById('local-model-progress-wrap');
      if (wrap) wrap.classList.add('hidden');
      // Cancel worker if running
      if (window.webclaw?.localML?.worker) {
        window.webclaw.localML.worker.postMessage({ type: 'CANCEL' });
        window.webclaw.localML.ready = false;
        window.webclaw.localML.worker = null;
      }
    });
  }

  // Poll for backend detection completion
  let backendPollCount = 0;
  const backendPoll = setInterval(() => {
    backendPollCount++;
    updateBackendDisplay();
    if (window.webclaw?.localML?.backend || backendPollCount > 20) {
      clearInterval(backendPoll);
    }
  }, 500);
})();
```
