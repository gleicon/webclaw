patSaveBtn.addEventListener("click", async function() {
  const val = patInput.value.trim();
  if (!val) return;

  if (!window.webclaw || !window.webclaw.oauth || !window.webclaw.oauth.savePATToken) {
    showToast("OAuth not ready", "error");
    return;
  }

  patSaveBtn.disabled = true;
  patSaveBtn.textContent = "Saving...";

  try {
    await window.webclaw.oauth.savePATToken(provider.id, val);
    patInput.value = "";
    patInput.type = "password";
    toggleBtn.textContent = "Show";
    showMaskedState(provider.id, statusSpan, patSection, patInputRow, updateBtn);
    showToast("Token saved for " + provider.name, "success");
  } catch (err) {
    console.error("[oauth] PAT save failed:", err);
    showToast("Failed to save token: " + (err.message || err), "error");
  } finally {
    patSaveBtn.disabled = false;
    patSaveBtn.textContent = "Save";
  }
});

**Step 5 — Write showMaskedState(providerId, statusSpan, patSection, patInputRow, updateBtn) helper:**

```javascript
function showMaskedState(providerId, statusSpan, patSection, patInputRow, updateBtn) {
  // Update status dot
  statusSpan.innerHTML = '<span class="inline-flex items-center"><span class="w-2 h-2 rounded-full bg-green-500 mr-1"></span>Connected</span>';
  statusSpan.className = "text-xs text-green-400";
  // Show masked container, hide input row
  const maskedDiv = document.getElementById("oauth-masked-" + providerId);
  if (maskedDiv) maskedDiv.classList.remove("hidden");
  if (patInputRow) patInputRow.classList.add("hidden");
  if (updateBtn) updateBtn.classList.remove("hidden");
}
```

Note: `statusSpan.innerHTML` here uses only trusted static strings (no variable user content), which is acceptable per CLAUDE.md (same pattern as the existing Connected Services status spans).

```javascript
if (status.connected) {
  if (status.valid === false) {
    // Invalid token — red dot state
    statusSpan.innerHTML = '<span class="inline-flex items-center"><span class="w-2 h-2 rounded-full bg-red-500 mr-1"></span>Invalid token</span>';
    statusSpan.className = "text-xs text-red-400";
    // Show re-enter link only for PAT providers
    const updateBtn = document.getElementById("oauth-pat-update-" + status.provider);
    if (updateBtn) {
      updateBtn.classList.remove("hidden");
      // Also show input row so user can re-enter
      const patInputRow = document.getElementById("oauth-pat-inputrow-" + status.provider);
      if (patInputRow) patInputRow.classList.remove("hidden");
    }
  } else if (status.auth_type === "pat") {
    // PAT connected — show masked state
    const maskedDiv = document.getElementById("oauth-masked-" + status.provider);
    const patInputRow = document.getElementById("oauth-pat-inputrow-" + status.provider);
    const updateBtn = document.getElementById("oauth-pat-update-" + status.provider);
    if (maskedDiv) maskedDiv.classList.remove("hidden");
    if (patInputRow) patInputRow.classList.add("hidden");
    statusSpan.innerHTML = '<span class="inline-flex items-center"><span class="w-2 h-2 rounded-full bg-green-500 mr-1"></span>Connected</span>';
    statusSpan.className = "text-xs text-green-400";
    if (updateBtn) updateBtn.classList.remove("hidden");
  } else {
    // OAuth connected
    const username = status.username ? " (" + status.username + ")" : "";
    statusSpan.innerHTML = '<span class="inline-flex items-center"><span class="w-2 h-2 rounded-full bg-green-500 mr-1"></span>Connected' + username + '</span>';
    statusSpan.className = "text-xs text-green-400";
    if (btn) {
      btn.textContent = "Disconnect";
      btn.className = "bg-red-700 hover:bg-red-600 text-white rounded px-3 py-1.5 text-xs font-medium";

  // Not connected
  statusSpan.innerHTML = '<span class="inline-flex items-center"><span class="w-2 h-2 rounded-full bg-gray-500 mr-1"></span>Not connected</span>';
  statusSpan.className = "text-xs text-gray-500";

    btn.textContent = "Connect";
    btn.className = "bg-indigo-600 hover:bg-indigo-500 text-white rounded px-3 py-1.5 text-xs font-medium";
  }
}
```
