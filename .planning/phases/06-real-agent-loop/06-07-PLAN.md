    ```go
    type Router struct {
        providers map[string]*ProviderChain  // Changed from Provider
        current   string
        retry     RetryConfig
    }

    // NewRouter creates router with default retry config
    func NewRouter() *Router {
        return &Router{
            providers: make(map[string]*ProviderChain),
            retry:     DefaultRetryConfig(),
        }
    }

    // SetProvider wraps provider in ProviderChain with retry
    func (r *Router) SetProvider(name string, p Provider) {
        // Get model ID from provider if available
        modelID := "default"
        if mp, ok := p.(interface{ GetModel() string }); ok {
            modelID = mp.GetModel()
        }
        
        // Wrap in ProviderChain
        chain := NewProviderChain(p, modelID)
        chain.retry = r.retry
        r.providers[name] = chain
    }

    // SetFallback configures fallback provider for a primary
    func (r *Router) SetFallback(primaryName string, fallback Provider, fallbackModel string) {
        if chain, ok := r.providers[primaryName]; ok {
            chain.SetFallback(fallback, fallbackModel)
        }
    }

    // Stream delegates to ProviderChain for automatic retries
    func (r *Router) Stream(ctx context.Context, modelID string, req CompletionRequest) (<-chan Token, error) {
        if r.current == "" {
            return nil, errors.New("no provider selected")
        }
        
        chain, ok := r.providers[r.current]
        if !ok {
            return nil, fmt.Errorf("provider '%s' not found", r.current)
        }
        
        // ProviderChain handles retries automatically
        return chain.Stream(ctx, req)
    }
    ```

    ```go
    func parseSSEEvent(data string) *SSEEvent {
        lines := strings.Split(data, "\n")
        for _, line := range lines {
            if strings.HasPrefix(line, "data: ") {
                payload := strings.TrimPrefix(line, "data: ")
                if payload == "[DONE]" {
                    return &SSEEvent{Done: true}
                }
                // Parse JSON payload
                var event map[string]interface{}
                json.Unmarshal([]byte(payload), &event)
                return &SSEEvent{Data: event}
            }
        }

    }
    ```

    ```go
        // Create mock provider that fails twice then succeeds
        mockProv := &mockRetryProvider{

            BackoffMultiplier: 2.0,

        req := provider.CompletionRequest{
            Model:    "test",
            Messages: []provider.Message{{Role: "user", Content: "hello"}},
            Stream:   true,
        }
        
        start := time.Now()
        ch, err := chain.Stream(ctx, req)
        if err != nil {
            t.Fatalf("unexpected error: %v", err)

        // Consume all tokens
        var tokens []provider.Token
        for tok := range ch {
            tokens = append(tokens, tok)

        // Should have succeeded after 2 retries
        if len(tokens) == 0 {
            t.Error("expected tokens, got none")

        if elapsed < 250*time.Millisecond {
            t.Errorf("expected backoff delay, elapsed: %v", elapsed)

        if mockProv.actualRetries != 2 {
            t.Errorf("expected 2 retries, got %d", mockProv.actualRetries)
        }
    ```
