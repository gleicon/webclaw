Anthropic tool_use format:
- Request: `tools` array with `name`, `description`, `input_schema`
- Response: `content_block_start` with `type: "tool_use"`, `id`, `name`, `input`

OpenAI tool_calls format:
- Request: `tools` array with `type: "function"`, `function.name`, `function.description`, `function.parameters`
- Response: `tool_calls` array with `id`, `type`, `function.name`, `function.arguments`

    ```go
        if streamEvent.ContentBlock != nil && streamEvent.ContentBlock.Type == "tool_use" {

            toolUseID = streamEvent.ContentBlock.ID
            toolName = streamEvent.ContentBlock.Name
            toolInputJSON.Reset()
        }
    ```

    4. Handle input_json_delta:
    ```go
    case "content_block_delta":
        if inToolUse && streamEvent.Delta != nil && streamEvent.Delta.Type == "input_json_delta" {

        } else if streamEvent.Delta != nil && streamEvent.Delta.Type == "text_delta" {
            tokenChan <- Token{Text: streamEvent.Delta.Text}
        }
    ```

    ```go
    case "message_stop":
        if inToolUse {
            // Parse accumulated JSON
            var toolInput map[string]interface{}
            json.Unmarshal([]byte(toolInputJSON.String()), &toolInput)
            
            tokenChan <- Token{
                FinishReason: "tool_use",
                ToolName:     toolName,
                ToolInput:    toolInput,
                ToolUseID:    toolUseID,
            }
        } else {
            tokenChan <- Token{FinishReason: "stop"}
        }
        return
    ```
