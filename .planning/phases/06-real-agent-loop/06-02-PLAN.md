    ```go
    func (pa *providerAdapter) Stream(
        ctx context.Context,
        messages []Message,
        tools []map[string]interface{},  // NEW PARAMETER
        callback func(tok provider.Token),
    ) error {
        // Check if provider is available
        hasProv := pa.router.HasProvider(pa.name)
        if !hasProv {
            return fmt.Errorf("provider '%s' not available", pa.name)
        }

        // Convert agent []Message to provider []Message
        provMsgs := make([]provider.Message, len(messages))
        for i, m := range messages {
            provMsgs[i] = provider.Message{Role: m.Role, Content: m.Content}
        }

        // Create request WITH tools
        req := provider.CompletionRequest{
            Model:       pa.model,
            Messages:    provMsgs,
            Tools:       tools,  // FORWARD TO PROVIDER
            MaxTokens:   4096,
            Temperature: 0.7,
            Stream:      true,
        }

        ch, err := pa.router.Stream(ctx, pa.model, req)
        if err != nil {
            return err
        }

        for tok := range ch {
            if tok.FinishReason == "error" {
                return fmt.Errorf("provider error: %s", tok.Text)
            }
            callback(tok)
        }
        return nil
    }
    ```

    Also update the Provider interface in this file if needed:
    ```go

        Stream(ctx context.Context, messages []Message, tools []map[string]interface{}, callback func(tok provider.Token)) error
        GetName() string
        GetModel() string
    }
    ```

    ```go
    package e2e

    import (
        "testing"
        "github.com/gleicon/webclaw/internal/agent"
        "github.com/gleicon/webclaw/internal/tools"
    )

    func TestToolFlow_RegistryToProvider(t *testing.T) {
        // Create tool registry with test tool
        registry := tools.NewRegistry()
        registry.Register(&tools.Tool{
            Name:        "test_echo",
            Description: "Echoes back the input",
            InputSchema: map[string]interface{}{
                "type": "object",
                "properties": map[string]interface{}{
                    "message": map[string]interface{}{"type": "string"},
                },
                "required": []string{"message"},
            },
            Execute: func(ctx context.Context, params map[string]interface{}) (*tools.ToolResult, error) {
                msg := params["message"].(string)
                return &tools.ToolResult{Content: msg}, nil
            },
        })

        // Verify ToAPISchema returns tools
        schemas := registry.ToAPISchema()
        if len(schemas) != 1 {
            t.Errorf("expected 1 tool schema, got %d", len(schemas))
        }

        // Verify schema has required fields
        schema := schemas[0]
        if schema["name"] != "test_echo" {
            t.Errorf("expected name 'test_echo', got %v", schema["name"])
        }
    }
    ```
