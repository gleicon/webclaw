    ```go
    // CheckQuota returns current storage usage
    func (e *LRUEvictor) CheckQuota(ctx context.Context) (*QuotaInfo, error) {
        // Call navigator.storage.estimate() via jsbridge
        estimate := js.Global().Get("navigator").Get("storage").Call("estimate")
        
        // Wait for promise
        var quota, usage int64
        promiseHandler := js.FuncOf(func(this js.Value, args []js.Value) interface{} {
            result := args[0]
            quota = int64(result.Get("quota").Float())
            usage = int64(result.Get("usage").Float())
            return nil
        })
        defer promiseHandler.Release()
        
        estimate.Call("then", promiseHandler)
        
        percent := float64(usage) / float64(quota)
        return &QuotaInfo{
            Usage:   usage,
            Quota:   quota,
            Percent: percent,
            ShouldEvict: percent >= e.targetRatio,
        }, nil
    }
    ```

    ```go
        // PHASE 6-6: Check quota before storing (MEM-05)
        if s.e victor != nil {
            quota, err := s.evictor.CheckQuota(context.Background())
            if err == nil && quota.ShouldEvict {
                js.Global().Get("console").Call("log",
                    "webclaw: storage at", int(quota.Percent*100), "%- triggering eviction")
                
                // Evict oldest memories
                if err := s.EvictIfNeeded(); err != nil {
                    js.Global().Get("console").Call("error",
                        "webclaw: eviction failed:", err.Error())
                }
            }
        }

        // Now proceed with store
        key := fmt.Sprintf("mem:%s", doc.ID)
        data, err := json.Marshal(doc)
        if err != nil {
            return err
        }
        
        return jsbridge.MemorySet(key, string(data))
    }
    ```
