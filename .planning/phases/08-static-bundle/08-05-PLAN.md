       // ImportFromJSON creates a Conversation from exported JSON data
       func ImportFromJSON(data []byte) (*Conversation, error) {
           var export struct {
               Version      string       `json:"version"`
               ExportedAt   time.Time    `json:"exported_at"`
               Conversation Conversation `json:"conversation"`
           }
           
           if err := json.Unmarshal(data, &export); err != nil {
               return nil, fmt.Errorf("failed to parse export JSON: %w", err)
           }
           
           // Validate version
           if export.Version != "1.0" {
               return nil, fmt.Errorf("unsupported export version: %s", export.Version)
           }
           
           // Restore conversation with updated timestamp
           conv := &export.Conversation
           conv.UpdatedAt = time.Now()
           
           // Ensure ID is unique (generate new if importing into existing session)
           if conv.ID == "" {
               conv.ID = generateMessageID() // or use a dedicated ID generator
           }
           
           return conv, nil
       }
       ```

       // Export conversation
       function exportConversation() {
           worker.postMessage({
               type: 'exportConversation'
           });
       }
       
       // Handle export response
       function handleExportResponse(response) {
           if (response.error) {
               showError('Export failed: ' + response.error);
               return;
           }
           
           const data = response.data;
           const jsonBlob = new Blob([data.json], { type: 'application/json' });
           const url = URL.createObjectURL(jsonBlob);
           
           // Trigger download
           const a = document.createElement('a');
           a.href = url;
           a.download = `webclaw-conversation-${new Date().toISOString().slice(0,10)}.json`;
           document.body.appendChild(a);
           a.click();
           document.body.removeChild(a);
           URL.revokeObjectURL(url);
           
           showToast(`Exported ${data.messageCount} messages`);
       }
       
       // Import conversation
       function importConversation(file) {
           const reader = new FileReader();
           reader.onload = (e) => {
               const json = e.target.result;
               worker.postMessage({
                   type: 'importConversation',
                   json: json
               });
           };
           reader.readAsText(file);
       }
       
       // Handle import response
       function handleImportResponse(response) {
           if (response.error) {
               showError('Import failed: ' + response.error);
               return;
           }
           
           const data = response.data;
           showToast(`Imported ${data.messageCount} messages`);
           
           // Reload chat display
           loadConversation(data.id);
       }
       
       // Wire up event listeners
       document.getElementById('export-btn').addEventListener('click', exportConversation);
       document.getElementById('import-btn').addEventListener('click', () => {
           document.getElementById('import-file').click();
       });
       document.getElementById('import-file').addEventListener('change', (e) => {
           if (e.target.files.length > 0) {
               importConversation(e.target.files[0]);
           }
       });
       ```
