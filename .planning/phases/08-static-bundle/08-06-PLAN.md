       package telemetry
       
       import "time"
       
       // ErrorLevel indicates severity
       type ErrorLevel string
       
       const (
           ErrorLevelWarning  ErrorLevel = "warning"
           ErrorLevelError    ErrorLevel = "error"
           ErrorLevelCritical ErrorLevel = "critical"
       )
       
       // ErrorRecord represents a captured error
       type ErrorRecord struct {
           ID        string                 `json:"id"`
           Timestamp time.Time              `json:"timestamp"`
           Level     ErrorLevel             `json:"level"`
           Category  string                 `json:"category"` // "provider", "tool", "agent", "system"
           Message   string                 `json:"message"`
           Stack     string                 `json:"stack,omitempty"`
           Context   map[string]interface{} `json:"context,omitempty"`
           UserAgent string                 `json:"user_agent,omitempty"`
       }
       
       // EventRecord represents a significant event (not necessarily an error)
       type EventRecord struct {
           ID        string                 `json:"id"`
           Timestamp time.Time              `json:"timestamp"`
           Category  string                 `json:"category"`
           Action    string                 `json:"action"`
           Duration  int64                  `json:"duration_ms,omitempty"`
           Context   map[string]interface{} `json:"context,omitempty"`
       }
       
       // Sanitize removes sensitive data from context
       func SanitizeContext(ctx map[string]interface{}) map[string]interface{} {
           sensitiveKeys := []string{"api_key", "token", "password", "secret", "key"}
           cleaned := make(map[string]interface{})
           for k, v := range ctx {
               lowerKey := strings.ToLower(k)
               isSensitive := false
               for _, sensitive := range sensitiveKeys {
                   if strings.Contains(lowerKey, sensitive) {
                       isSensitive = true
                       break
                   }
               }
               if isSensitive {
                   cleaned[k] = "[REDACTED]"
               } else {
                   cleaned[k] = v
               }
           }
           return cleaned
       }
       ```

       package telemetry
       
       import (
           "context"
           "fmt"
           "runtime/debug"
           "strings"
           "time"
       )
       
       // Config holds telemetry configuration
       type Config struct {
           Enabled          bool   // Enable/disable telemetry
           MaxErrorsStored  int    // Maximum errors to keep (default: 100)
           MaxEventsStored  int    // Maximum events to keep (default: 50)
           ReportErrors     bool   // Include in user reports
       }
       
       // DefaultConfig returns sensible defaults
       func DefaultConfig() Config {
           return Config{
               Enabled:         true,
               MaxErrorsStored: 100,
               MaxEventsStored: 50,
               ReportErrors:    true,
           }
       }
       
       // Collector is the main telemetry interface
       type Collector struct {
           config  Config
           storage Storage
       }
       
       // Storage interface for persistence
       type Storage interface {
           StoreError(ctx context.Context, err ErrorRecord) error
           StoreEvent(ctx context.Context, evt EventRecord) error
           GetErrors(ctx context.Context, limit int) ([]ErrorRecord, error)
           GetEvents(ctx context.Context, limit int) ([]EventRecord, error)
           ClearOld(ctx context.Context, maxAge time.Duration) error
       }
       
       // NewCollector creates a telemetry collector
       func NewCollector(config Config, storage Storage) *Collector {
           return &Collector{
               config:  config,
               storage: storage,
           }
       }
       
       // RecordError captures an error with full context
       func (c *Collector) RecordError(level ErrorLevel, category string, err error, context map[string]interface{}) {
           if !c.config.Enabled {
               return
           }
           
           record := ErrorRecord{
               ID:        generateID(),
               Timestamp: time.Now(),
               Level:     level,
               Category:  category,
               Message:   err.Error(),
               Stack:     string(debug.Stack()),
               Context:   SanitizeContext(context),
           }
           
           // Async store to not block execution
           go func() {
               ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
               defer cancel()
               
               if err := c.storage.StoreError(ctx, record); err != nil {
                   // If telemetry fails, just log to console (don't cascade)
                   fmt.Printf("[Telemetry] Failed to store error: %v\n", err)
               }
           }()
       }
       
       // RecordEvent captures a significant event
       func (c *Collector) RecordEvent(category, action string, duration int64, context map[string]interface{}) {
           if !c.config.Enabled {
               return
           }
           
           record := EventRecord{
               ID:        generateID(),
               Timestamp: time.Now(),
               Category:  category,
               Action:    action,
               Duration:  duration,
               Context:   SanitizeContext(context),
           }
           
           go func() {
               ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
               defer cancel()
               
               if err := c.storage.StoreEvent(ctx, record); err != nil {
                   fmt.Printf("[Telemetry] Failed to store event: %v\n", err)
               }
           }()
       }
       
       // GetReport generates a telemetry report for debugging
       func (c *Collector) GetReport(ctx context.Context) (*TelemetryReport, error) {
           errors, err := c.storage.GetErrors(ctx, 50)
           if err != nil {
               return nil, fmt.Errorf("failed to get errors: %w", err)
           }
           
           events, err := c.storage.GetEvents(ctx, 20)
           if err != nil {
               return nil, fmt.Errorf("failed to get events: %w", err)
           }
           
           return &TelemetryReport{
               GeneratedAt: time.Now(),
               Errors:      errors,
               Events:      events,
           }, nil
       }
       
       // TelemetryReport is a snapshot for export
       type TelemetryReport struct {
           GeneratedAt time.Time      `json:"generated_at"`
           Errors      []ErrorRecord  `json:"errors"`
           Events      []EventRecord  `json:"events"`
       }
       
       func generateID() string {
           return fmt.Sprintf("tel_%d_%s", time.Now().UnixNano(), randomString(6))
       }
       
       func randomString(length int) string {
           const charset = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
           result := make([]byte, length)
           for i := range result {
               result[i] = charset[time.Now().UnixNano()%int64(len(charset))]
           }
           return string(result)
       }
       ```

       // StoreError saves an error record to IndexedDB
       func (s *IndexedDBStorage) StoreError(ctx context.Context, err ErrorRecord) error {
           data, err := json.Marshal(err)
           if err != nil {
               return fmt.Errorf("marshal error: %w", err)
           }
           
           return s.storeItem(ctx, "errors", err.ID, data)
       }
       
       // StoreEvent saves an event record to IndexedDB
       func (s *IndexedDBStorage) StoreEvent(ctx context.Context, evt EventRecord) error {
           data, err := json.Marshal(evt)
           if err != nil {
               return fmt.Errorf("marshal event: %w", err)
           }
           
           return s.storeItem(ctx, "events", evt.ID, data)
       }
       
       // GetErrors retrieves recent errors
       func (s *IndexedDBStorage) GetErrors(ctx context.Context, limit int) ([]ErrorRecord, error) {
           items, err := s.getItems(ctx, "errors", limit)
           if err != nil {
               return nil, err
           }
           
           errors := make([]ErrorRecord, 0, len(items))
           for _, item := range items {
               var err ErrorRecord
               if err := json.Unmarshal(item, &err); err != nil {
                   continue // Skip corrupted records
               }
               errors = append(errors, err)
           }
           return errors, nil
