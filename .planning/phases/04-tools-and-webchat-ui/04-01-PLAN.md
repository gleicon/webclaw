The agent `Provider` interface in loop.go requires: `Stream(ctx, []Message, func(token string)) error`, `GetName() string`, `GetModel() string`. The `provider.Router` has a different Stream signature (returns channel, not callback). Create an adapter in loop.go:

```go
// providerAdapter wraps provider.Router to match the agent.Provider interface
type providerAdapter struct {
    router *provider.Router
    name   string
    model  string
}

func (pa *providerAdapter) Stream(ctx context.Context, messages []Message, callback func(token string)) error {
    // Convert agent []Message to provider []Message
    provMsgs := make([]provider.Message, len(messages))
    for i, m := range messages {
        provMsgs[i] = provider.Message{Role: m.Role, Content: m.Content}
    }
    req := provider.CompletionRequest{
        Model:       pa.model,
        Messages:    provMsgs,
        MaxTokens:   4096,
        Temperature: 0.7,
        Stream:      true,
    }
    ch, err := pa.router.Stream(ctx, pa.model, req)
    if err != nil {
        return err
    }
    for tok := range ch {
        if tok.FinishReason == "error" {
            return fmt.Errorf("provider error: %s", tok.Text)
        }
        if tok.Text != "" {
            callback(tok.Text)
        }
    }
    return nil
}

func (pa *providerAdapter) GetName() string { return pa.name }
func (pa *providerAdapter) GetModel() string { return pa.model }
```
