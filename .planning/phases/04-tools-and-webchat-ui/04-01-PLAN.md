The agent `Provider` interface in loop.go requires: `Stream(ctx, []Message, func(token string)) error`, `GetName() string`, `GetModel() string`. The `provider.Router` has a different Stream signature (returns channel, not callback). Create an adapter in loop.go:

```go
// providerAdapter wraps provider.Router to match the agent.Provider interface
type providerAdapter struct {
    router *provider.Router
    name   string
    model  string
}

func (pa *providerAdapter) Stream(ctx context.Context, messages []Message, callback func(token string)) error {
    // Convert agent []Message to provider []Message
    provMsgs := make([]provider.Message, len(messages))
    for i, m := range messages {
        provMsgs[i] = provider.Message{Role: m.Role, Content: m.Content}
    }
    req := provider.CompletionRequest{
        Model:       pa.model,
        Messages:    provMsgs,
        MaxTokens:   4096,
        Temperature: 0.7,
        Stream:      true,
    }
    ch, err := pa.router.Stream(ctx, pa.model, req)
    if err != nil {
        return err
    }
    for tok := range ch {
        if tok.FinishReason == "error" {
            return fmt.Errorf("provider error: %s", tok.Text)
        }
        if tok.Text != "" {
            callback(tok.Text)
        }
    }
    return nil
}

func (pa *providerAdapter) GetName() string { return pa.name }
func (pa *providerAdapter) GetModel() string { return pa.model }
```

Extend `AgentLoop.Run()` to handle LLM tool_use responses. This is the missing piece that makes TOOL-01 through TOOL-04 achievable end-to-end.

The dispatch loop must:

1. **Detect tool_use responses**: After the streaming call completes, inspect the accumulated tokens. When `provider.Token.FinishReason == "tool_use"`, the provider populates `Token.ToolName`, `Token.ToolInput`, and `Token.ToolUseID` on the final token. Capture these fields from the channel.

   NOTE: The current `Provider` interface callback pattern `func(token string)` does not carry structured tool data. To thread tool_use metadata through the adapter, extend the callback or change the approach. Recommended: change the `Provider` interface callback to `func(tok provider.Token)` so the full Token (including FinishReason, ToolName, ToolInput, ToolUseID) is available in the loop. Update `providerAdapter.Stream` (added in Task 3) accordingly. Update `mockProvider.Stream` to pass the final token with FinishReason="stop" for backward compatibility.

2. **Emit "running" event**: Before dispatching, call `al.workerBridge.EmitToolEvent(toolName, "running", "Running "+toolName+"...", "")`.

3. **Dispatch through registry**: Call `al.toolRegistry.Dispatch(ctx, toolName, toolInput)`.
   - If `al.toolRegistry == nil`, log a warning and inject a tool_result error message: `"tool registry not configured"`.
   - On dispatch error: call `al.workerBridge.EmitToolEvent(toolName, "error", err.Error(), err.Error())` and inject a tool_result with `isError: true`.

4. **Emit "done" or "error" event**: After dispatch, call `al.workerBridge.EmitToolEvent(toolName, result.Status, result.DisplayContent, result.Content)`.

5. **Inject tool_result into message history**: Append two messages:
   - The assistant message containing the tool_use block (Anthropic format): `{"role": "assistant", "content": [{"type": "tool_use", "id": toolUseID, "name": toolName, "input": toolInput}]}`
   - The user message containing the tool_result block: `{"role": "user", "content": [{"type": "tool_result", "tool_use_id": toolUseID, "content": result.Content, "is_error": result.IsError}]}`
   Since the existing `Message` type uses `Content string`, format the content as JSON strings using `encoding/json`. Alternatively, if the provider/router already handles structured content blocks, follow its convention.

6. **Loop up to maxToolIterations**: Define `const maxToolIterations = 10` in loop.go. Track iteration count. If FinishReason is NOT "tool_use" (i.e., "stop", "length", etc.), break the loop and return normally. If the iteration limit is reached, break and append a user-visible message: "Tool iteration limit reached."

7. **Guard on nil workerBridge**: If `al.workerBridge == nil`, skip EmitToolEvent calls silently (tool execution still works, just no UI events).

**Pseudocode for the dispatch loop in Run():**

```go
const maxToolIterations = 10

func (al *AgentLoop) Run(ctx context.Context, userMessage string) error {
    al.messages = append(al.messages, Message{Role: "user", Content: userMessage})

    for iter := 0; iter < maxToolIterations; iter++ {
        provider, err := al.getProvider()
        // ... error handling ...

        var lastToken provider.Token  // capture final token with FinishReason
        var tokenBuf strings.Builder

        err = provider.Stream(ctx, al.messages, func(tok provider.Token) {
            if tok.Text != "" {
                tokenBuf.WriteString(tok.Text)
                al.onToken(tok.Text)  // stream to UI
            }
            lastToken = tok  // always capture; last one has FinishReason
        })
        if err != nil {
            return err
        }

        if lastToken.FinishReason != "tool_use" {
            // Normal completion — append assistant message and return
            al.messages = append(al.messages, Message{Role: "assistant", Content: tokenBuf.String()})
            al.onComplete()
            return nil
        }

        // Tool use detected
        toolName := lastToken.ToolName
        toolInput := lastToken.ToolInput
        toolUseID := lastToken.ToolUseID

        if al.workerBridge != nil {
            al.workerBridge.EmitToolEvent(toolName, "running", "Running "+toolName+"...", "")
        }

        var result *tools.ToolResult
        if al.toolRegistry != nil {
            result, err = al.toolRegistry.Dispatch(ctx, toolName, toolInput)
        } else {
            result = &tools.ToolResult{IsError: true, Content: "tool registry not configured",
                DisplayContent: "tool registry not configured", ToolName: toolName, Status: "error"}
        }

        if err != nil {
            result = &tools.ToolResult{IsError: true, Content: err.Error(),
                DisplayContent: err.Error(), ToolName: toolName, Status: "error"}
        }

        if al.workerBridge != nil {
            al.workerBridge.EmitToolEvent(toolName, result.Status, result.DisplayContent, result.Content)
        }

        // Inject tool_use + tool_result into message history
        // (format as JSON strings for the existing string-Content Message type)
        toolUseJSON, _ := json.Marshal(map[string]interface{}{
            "type": "tool_use", "id": toolUseID, "name": toolName, "input": toolInput,
        })
        toolResultJSON, _ := json.Marshal(map[string]interface{}{
            "type": "tool_result", "tool_use_id": toolUseID,
            "content": result.Content, "is_error": result.IsError,
        })
        al.messages = append(al.messages,
            Message{Role: "assistant", Content: string(toolUseJSON)},
            Message{Role: "user", Content: string(toolResultJSON)},
        )
        // Loop: next iteration sends updated history back to LLM
    }

    // Iteration limit reached
    al.onComplete()
    return fmt.Errorf("tool iteration limit (%d) reached", maxToolIterations)
}
```
